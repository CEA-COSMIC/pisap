[docs]class SoftThreshold2Dt(object):
    """ Threshold proximity operator

    This class defines the threshold proximity operator

    Parameters
    ----------
    weights : np.ndarray
        Input array of weights
    """

    def __init__(self, weights, grad_op, linear_op, verbose=0):
        """ Initialize the SoftThreshold2Dt class.

        Parameters
        ----------
        grad_op: instance
            Gradient operator.
        linear_op: instance
            Linear operator.
        verbose: int, default 0
            Control the verbosity level.
        """
        self.grad_op = grad_op
        self.linear_op = linear_op
        self.update_weights(weights)
        self.verbose = verbose

[docs]    def op(self, data, extra_factor=1.0):
        """ Operator

        This method returns the input data thresholded by the weights

        Parameters
        ----------
        data : np.ndarray
            Input data array
        extra_factor : float
            Additional multiplication factor

        Returns
        -------
        np.ndarray thresholded data
        """
        std_est = multiscale_sigma_mad(self.grad_op, self.linear_op)
        if self.verbose > 0:
            print("[info] Proximity variance estimate: {0}.".format(std_est))
        data_at_scale = []
        nb_scale = self.linear_op.transform.nb_scale
        scale_last_index = self.linear_op.transform.scales_padds[nb_scale - 1]
        std_data = numpy.zeros((scale_last_index, ))
        for scale in range(nb_scale - 1):
            scale_indices = (
                self.linear_op.transform.scales_padds[scale],
                self.linear_op.transform.scales_padds[scale + 1])
            std_data[scale_indices[0]: scale_indices[1]] = std_est[scale]
            # TODO implement [:-1] on transform
            data_at_scale.append(data[scale_indices[0]: scale_indices[1]])
        data_at_scale = numpy.concatenate(data_at_scale)
        weights = self.weights[:scale_last_index]
        threshold = weights * std_data
        data_at_scale = denoise(data_at_scale, threshold, "soft")
        scale_indices = (
                self.linear_op.transform.scales_padds[nb_scale - 1],
                self.linear_op.transform.scales_padds[nb_scale])
        data_coarse_scale = data[scale_indices[0]: scale_indices[1]]
        return numpy.concatenate((data_at_scale, data_coarse_scale))
